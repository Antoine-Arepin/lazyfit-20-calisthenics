## Core Implementation

### 1. Main App (App.jsx)
```jsx
import { BrowserRouter, Routes, Route } from 'react-router-dom'
import { Header, Navigation } from './components/layout'
import { Home, Workout, Social, Progress, Profile, Settings } from './pages'

function App() {
  return (
    <BrowserRouter>
      <div className="min-h-screen bg-gray-900 text-white">
        <Header />
        <main className="container mx-auto px-4 py-8">
          <Routes>
            <Route path="/" element={<Home />} />
            <Route path="/workout" element={<Workout />} />
            <Route path="/social" element={<Social />} />
            <Route path="/progress" element={<Progress />} />
            <Route path="/profile" element={<Profile />} />
            <Route path="/settings" element={<Settings />} />
          </Routes>
        </main>
        <Navigation />
      </div>
    </BrowserRouter>
  )
}
export default App
```

### 3. Gamification Store (store/gamificationStore.js)
```javascript
import { create } from 'zustand'
import { persist } from 'zustand/middleware'

const useGamificationStore = create(
  persist(
    (set, get) => ({
      level: 1,
      xp: 0,
      achievements: [],
      dailyQuests: [],
      streak: 0,
      
      addXP: (amount) => set((state) => {
        const newXP = state.xp + amount
        const newLevel = Math.floor(newXP / 1000) + 1
        return { xp: newXP, level: newLevel }
      }),
      
      unlockAchievement: (achievementId) => set((state) => ({
        achievements: [...state.achievements, {
          id: achievementId,
          unlockedAt: new Date().toISOString()
        }]
      })),
      
      completeQuest: (questId) => set((state) => ({
        dailyQuests: state.dailyQuests.map(q => 
          q.id === questId ? { ...q, completed: true } : q
        )
      }))
    }),
    { name: 'lazyfit-v3-gamification' }
  )
)

export default useGamificationStore
```

### 4. Social Store (store/socialStore.js)
```javascript
import { create } from 'zustand'

const useSocialStore = create((set) => ({
  feed: [],
  challenges: [],
  leaderboard: [],
  friends: [],
  
  addPost: (post) => set((state) => ({
    feed: [{ ...post, id: Date.now(), timestamp: new Date() }, ...state.feed]
  })),
  
  joinChallenge: (challengeId) => set((state) => ({
    challenges: state.challenges.map(c => 
      c.id === challengeId ? { ...c, participants: c.participants + 1 } : c
    )
  }))
}))

export default useSocialStore
```

## Key Features Implementation

### Workout Builder Component
```jsx
import { useState } from 'react'
import { EXERCISES_DB } from '../utils/exercises'

export function WorkoutBuilder() {
  const [selected, setSelected] = useState([])
  
  const toggleExercise = (exercise) => {
    setSelected(prev => 
      prev.includes(exercise)
        ? prev.filter(e => e !== exercise)
        : [...prev, exercise]
    )
  }
  
  return (
    <div className="grid grid-cols-2 gap-4">
      {EXERCISES_DB.map(ex => (
        <button
          key={ex.id}
          onClick={() => toggleExercise(ex)}
          className={`p-4 rounded-lg ${
            selected.includes(ex) ? 'bg-blue-600' : 'bg-gray-800'
          }`}
        >
          {ex.name}
        </button>
      ))}
    </div>
  )
}
```

### Level System Component
```jsx
import useGamificationStore from '../store/gamificationStore'

export function LevelSystem() {
  const { level, xp } = useGamificationStore()
  const nextLevelXP = level * 1000
  const progress = (xp % 1000) / 1000 * 100
  
  return (
    <div className="bg-gradient-to-r from-purple-600 to-blue-600 p-6 rounded-xl">
      <div className="flex items-center justify-between mb-4">
        <h3 className="text-2xl font-bold">Level {level}</h3>
        <span className="text-sm">{xp} / {nextLevelXP} XP</span>
      </div>
      <div className="h-4 bg-gray-800 rounded-full overflow-hidden">
        <div 
          className="h-full bg-gradient-to-r from-yellow-400 to-orange-500"
          style={{ width: `${progress}%` }}
        />
      </div>
    </div>
  )
}
```
